DEFINE_IMPORT(FileName=C:\DE03 SYRINGE SCRIPT APP\RECIPES\COMMONPOINTS.TXT)


DEFINE_VARIABLE(VariableName=PIEZO_INBURST_INSP, _
 VariableType=INT, _
 VariableValue=10000)


DEFINE_VARIABLE(VariableName=PIEZO_INBURST_ONGRID, _
 VariableType=INT, _
 VariableValue=1)


DEFINE_VARIABLE(VariableName=I1, _
 VariableType=INT, _
 VariableValue=1)


'SET TO 1 IF YOU WANT TO FIRE WITH TRAPEZOID ELSE COS
DEFINE_VARIABLE(VariableName=USE_TRAPEZOID, _
 VariableType=INT, _
 VariableValue=1)



' LETS GO SLOW FOR NOW


DEFINE_VARIABLE(VariableName=FULL_SPEED, _
 VariableType=INT, _
 VariableValue=100)



' CYLINDER NOT ROTATED


IO_SET_OUTPUT(IONumber=OUT_ROTATE_OFF)


IO_SET_OUTPUT(IONumber=OUT_ETHANE_DOWN)


MOVE_TO_POINT(Tool=SYSTEM_TIP1, _
 Point=INS_T1_PT, _
 XOffset=0, _
 YOffset=0, _
 ZOffset=0, _
 Speed_pct=FULL_SPEED, _
 MoveToSafeHeightFirst=TRUE)


GET_INTEGER_FROM_USER(VariableName=I1, _
 Prompt= ENTER NUMBER OF PRIME CYCLES (TURN RESERVOIR VACUUM ON) 0 TO CANCEL, _
 MinValue=0, _
 MaxValue=10, _
 ValueIfUserCancels=0)


IF(I1 > 0)

PRIME(SyringeMask=0, _
 VolumePerStroke_uL=100, _
 NumSyringeStrokes=I1, _
 SyringeSpeedPull=11, _
 SyringeSpeedPush=17, _
 DelayAfter_ms=500, _
 Tip1Point=INS_T1_PT, _
 XOffset=0, _
 YOffset=0, _
 ZOffset=0, _
 MoveSpeedIn_pct=20, _
 MoveSpeedOut_pct=20)
	
ENDIF


MOVE_TO_POINT(Tool=SYSTEM_TIP1, _
 Point=INS_T1_PT, _
 XOffset=0, _
 YOffset=0, _
 ZOffset=0, _
 Speed_pct=FULL_SPEED, _
 MoveToSafeHeightFirst=TRUE)



SYRINGE_SET_VALVE(SyringeMask=0, _
ValvePosition=2, _
DelayAfter_ms=100)



VARIABLE_SET(VariableName=I1, _
 VariableValue=1)



GET_INTEGER_FROM_USER(VariableName=I1, _
Prompt=0=END  1=FIRE PIEZO, _
MinValue=0, _
MaxValue=1, _
ValueIfUserCancels=0)


IF(I1 = 0)


	EXITSEQUENCE


ENDIF


WHILE(I1 = 1)



	IF (USE_TRAPEZOID=1)
		 DE03_TRAP_SETUP(Tip=1, _
	 	 Leading=3, _
	 	 Dwell=15, _
	 	 Trailing=10, _
	 	 TrapDrops=500, _
		 TrapFreq=500, _
		 TrapAmp=600, _
		 StrobeDuration=1, _
		 StrobeDelay=100, _
	 	 TriggerSetting=0, _
	 	 TriggerDelay=1000000, _
	 	 TriggerPeriod=1)
	ELSE
		DE03_COS_SETUP(Tip=1, _
	 	 DesiredFreq=12750, _
	 	 InBurstNo=PIEZO_INBURST_INSP, _
	 	 CosAmplitude=450, _
	 	 StrobeDuration=1, _
	 	 StrobeDelay=0, _
	 	 NumOfBursts=1, _
	 	 TriggerSetting=0, _
	 	 TriggerDelay=1000000, _
	 	 TriggerPeriod=1)
	ENDIF



	DE03_START_WAVEFORM(WaitForCompletion=FALSE, _
	 Timeout_ms=0)


         GET_INTEGER_FROM_USER(VariableName=I1, _
	 Prompt=0=ABORT  1=FIRE PIEZO, _
	 MinValue=0, _
	 MaxValue=1, _
	 ValueIfUserCancels=0)


	
LOOP


' CANCEL WAS CHOSEN BY USER
